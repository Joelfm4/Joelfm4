struct Joel {
    char* languages[6];
    char* name;
    int age;
};

int main() {
    struct Joel joel = {
        .languages = {"Python", "C", "SQL", "Java Script", "Java", "BASIC"},
        .name = "Joel",
        .age = 19
    };

    printf("Name: %s\n", joel.name);
    printf("Age: %d\n", joel.age);
    printf("Languages:\n");
    for (int i = 0; i < 6; i++) {
        printf("  %s\n", joel.languages[i]);
    }

    return 0;
}
